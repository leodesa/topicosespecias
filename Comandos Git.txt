git --version = Verifica a versão do git
git init = Inicia um repositorio vazio na pasta atual
git config --global user.name "Leonardo Sá" = Criar usuario no git
git config --global user.email "leonardo231198@gmail.com" = Cria email no usuario do git
git status = Verifica tudo que não foi salvo
git add . = Coloca os arquivos na área de transferencia
git commit -m "Mensagem" = Criar anotação do commit
git log = Mostras todo o log dos commits realizados
git remote add origin https://github.com/leodesa/topicosespecias.git = Setar endereço do GIT
git push -u origin master = Comitando pela primeira vez
